import time
import random
import matplotlib.pyplot as plt
import pandas as pd

#  
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

#    
sizes = [5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000, 10000, 20000, 50000, 100000]
insert_times, bubble_times = [], []

for size in sizes:
    data = [random.randint(0, 1000000) for _ in range(size)]

    #    (  2000,  )
    if size <= 2000:
        start = time.time()
        insertion_sort(data.copy())
        insert_times.append(time.time() - start)
    else:
        insert_times.append(None)  # ,  

    #    (  2000)
    if size <= 2000:
        start = time.time()
        bubble_sort(data.copy())
        bubble_times.append(time.time() - start)
    else:
        bubble_times.append(None)

#  
df = pd.DataFrame({
    "n": sizes,
    "T_insert(n)": insert_times,
    "T_bubble(n)": bubble_times
})
